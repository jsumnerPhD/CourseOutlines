\entry{Course content}{
The material to be covered is contained in the following chapters and sections of {\bf Physics for Scientists and Engineers by Knight, 3rd edition}.
\begin{center}
  \begin{tabulary}{0.95\textwidth}{|c|L|L|}
    \hline
    Weeks & Topics & Chapter \& Section \\ \hline\hline
   1 & Concepts of motion & Ch.1: 1--8\\ \hline
	2--3 & Kinematics in one dimension & Ch.2: 1--7\\ \hline
	4--5 & Motion in two or three dimensions (including circular motion) & Ch.3: 1--4; Ch.4: 1--7; Ch.12: 1\\ \hline
	6--7 & Dynamics in one dimension & Ch.5: 1--7; Ch.6: 1--4 (5 optional)\\ \hline
	8 & Newton's laws & Ch.7: 1--5\\ \hline
	9--10 & Dynamics in two dimensions & Ch.8: 1--5\\ \hline
	11 & Impulse and momentum	 & Ch.9: 1--6\\ \hline
	12--13 & Energy	 & Ch.10: 1--7\\ \hline
	14 & Work	& Ch.11: 1--6, 8, 9 (7 optional)\\ \hline
	15 & Rotation of a rigid body	& Ch.12: 2--7 (8--11 optional)\\ \hline
  \end{tabulary}
\end{center}
